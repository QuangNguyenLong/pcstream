106 {
107 return (pcs_vec3f_t){left.y * right.z - left.z * right.y,

111 static inline float pcs_vec3f_angle_between(pcs_vec3f_t left,
112 pcs_vec3f_t right)
113 {
114 float dot = pcs_vec3f_dot(left, right);
115 float magA = pcs_vec3f_magnitude(left);
116 float magB = pcs_vec3f_magnitude(right);
117 return (float)acos(dot / (magA * magB));
118 }

126 static inline int pcs_vec3f_g(pcs_vec3f_t left, pcs_vec3f_t right)
127 {
128 if (left.x < right.x)
129 {
130 return 0;
131 }
132 if (left.x > right.x)
133 {
134 return 1;
135 }
136 if (left.y < right.y)
137 {
138 return 0;
139 }
140 if (left.y > right.y)
141 {
142 return 1;
143 }
144 return left.z > right.z;
145 }
146 static inline int pcs_vec3f_l(pcs_vec3f_t left, pcs_vec3f_t right)

148 if (left.x < right.x)

150 return 1;

153 {

156 if (left.y < right.y)

158 return 1;
159 }

162 return 0;
163 }

190 static inline pcs_vec3f_t pcs_vec3f_mvp_mul(pcs_vec3f_t vec,
191 const float *mvp)
192 {
193 float temp_x =
194 mvp[0] * vec.x + mvp[4] * vec.y + mvp[8] * vec.z + mvp[12];
195 float temp_y =
196 mvp[1] * vec.x + mvp[5] * vec.y + mvp[9] * vec.z + mvp[13];
197 float temp_z =
198 mvp[2] * vec.x + mvp[6] * vec.y + mvp[10] * vec.z + mvp[14];
199 float temp_w =
200 mvp[3] * vec.x + mvp[7] * vec.y + mvp[11] * vec.z + mvp[15];
201 temp_x /= temp_w;
202 temp_y /= temp_w;
203 temp_z /= temp_w;
204 return (pcs_vec3f_t){temp_x, temp_y, temp_z};
205 }
206 static inline pcs_vec3f_t
207 pcs_vec3f_rotate(pcs_vec3f_t vec, float angle, pcs_vec3f_t axis)
208 {
209 float cosa = cosf(angle);
210 float sina = sinf(angle);
211 float one_minus_c = 1.0F - cosa;
212
213 float len =
214 sqrtf(axis.x * axis.x + axis.y * axis.y + axis.z * axis.z);
215 axis.x /= len;
216 axis.y /= len;
217 axis.z /= len;
218
219 float tmp[3][3] = {
220 { axis.x * axis.x * one_minus_c + cosa,
221 axis.x * axis.y * one_minus_c - axis.z * sina,
222 axis.x * axis.z * one_minus_c + axis.y * sina},
223 {axis.y * axis.x * one_minus_c + axis.z * sina,
224 axis.y * axis.y * one_minus_c + cosa,
225 axis.y * axis.z * one_minus_c - axis.x * sina},
226 {axis.z * axis.x * one_minus_c - axis.y * sina,
227 axis.z * axis.y * one_minus_c + axis.x * sina,
228 axis.z * axis.z * one_minus_c + cosa }
229 };
230
231 pcs_vec3f_t result;
232 result.x =
233 tmp[0][0] * vec.x + tmp[0][1] * vec.y + tmp[0][2] * vec.z;
234 result.y =
235 tmp[1][0] * vec.x + tmp[1][1] * vec.y + tmp[1][2] * vec.z;
236 result.z =
237 tmp[2][0] * vec.x + tmp[2][1] * vec.y + tmp[2][2] * vec.z;
238
239 return result;
240 }
