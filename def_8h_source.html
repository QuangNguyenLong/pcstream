27#define PCSTREAM_WRAP_ACTION(action) \
28 do \
29 { \
30 action; \
31 } while (0)
32
33#define PCSTREAM_ACTION_EXIT() \
34 PCSTREAM_WRAP_ACTION(exit(EXIT_FAILURE))
35
36#define PCSTREAM_ACTION_LOG_ERROR() \
37 PCSTREAM_WRAP_ACTION( \
38 fprintf(stderr, \
39 "[PCSTREAM ERROR] failed at %s:%d\n", \
40 __FILE__, \
41 __LINE__))
42
43#define PCSTREAM_ACTION_EXIT_ERROR() \
44 PCSTREAM_WRAP_ACTION({ \
45 PCSTREAM_ACTION_LOG_ERROR(); \
46 PCSTREAM_ACTION_EXIT(); \
47 })

49#define PCSTREAM_CHECK(expr, action) \
50 PCSTREAM_WRAP_ACTION({ \
51 if ((expr) != PCSTREAM_RET_SUCCESS) \
52 { \
53 action; \
54 } \
55 })

57#define PCSTREAM_CHECK_ERROR(expr) \
58 PCSTREAM_CHECK(expr, PCSTREAM_ACTION_LOG_ERROR())
